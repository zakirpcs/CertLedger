Generate a CSR with OpenSSL
# Generate private key and CSR
openssl req -new -newkey rsa:2048 -nodes \
  -keyout server.key \
  -out server.csr \
  -subj "/CN=your.domain.com/O=YourOrg/C=US"

# With Subject Alternative Names (recommended)
openssl req -new -newkey rsa:2048 -nodes \
  -keyout server.key \
  -out server.csr \
  -config <(cat /etc/ssl/openssl.cnf \
    <(printf "\n[SAN]\nsubjectAltName=DNS:your.domain.com,DNS:www.your.domain.com"))
